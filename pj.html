cap = cv2.VideoCapture(r'C:\Users\1\Desktop\shorts.mp4')

fourcc = cv2.VideoWriter_fourcc(*'FMP4')

width = cap.get(cv2.CAP_PROP_FRAME_WIDTH)
height = cap.get(cv2.CAP_PROP_FRAME_HEIGHT)
fps = cap.get(cv2.CAP_PROP_FPS)
delay = 6560

out = cv2.VideoWriter(r'C:\Users\1\Desktop\shorts.mp4(2).mp4', fourcc, fps, (int(width), int(height)))

while True:
    ret, frame = cap.read()

    if ret:
        detector_hog = dlib.get_frontal_face_detector()
        img_rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
        dlib_rects = detector_hog(img_rgb, 1)

        landmark_predictor = dlib.shape_predictor('shape_predictor_68_face_landmarks.dat')
        list_landmarks = []

        for dlib_rect in dlib_rects:
            points = landmark_predictor(img_rgb, dlib_rect)
            list_points = list(map(lambda p: (p.x, p.y), points.parts()))
            list_landmarks.append(list_points)

        for landmark in list_landmarks:
            eye_landmarks = landmark[37:46] + landmark[29:30] + landmark[16:17] + landmark[27:28] + landmark[1:2]
            min_x = min(eye_landmarks, key=lambda p: p[0])[0]
            min_y = min(eye_landmarks, key=lambda p: p[1])[1]
            max_x = max(eye_landmarks, key=lambda p: p[0])[0]
            max_y = max(eye_landmarks, key=lambda p: p[1])[1]
            width = max_x - min_x
            height = max_y - min_y
            img_bg = cv2.bitwise_and(img_rgb[min_y:max_y, min_x:max_x, :], cv2.cvtColor(cv2.resize(img_binary, (width, height)), cv2.COLOR_GRAY2BGR))
            resized_image = cv2.resize(img_bg, (width, height))
            resize_image = cv2.resize(img_tg, (width, height))
            img_rgb[min_y:max_y, min_x:max_x, :] = cv2.add(resized_image, resize_image)

        out.write(cv2.cvtColor(img_rgb, cv2.COLOR_RGB2BGR))

        cv2.imshow("VideoFrame",cv2.cvtColor(img_rgb, cv2.COLOR_RGB2BGR) )

        if cv2.waitKey(1) & 0xFF == ord('q'):
            break
    else:
        print("Can't receive frame. Exiting...")
        break

out.release()
cap.release()
cv2.destroyAllWindows()
